Copy the exact server block below and paste it directly into your Apps Script editor file (Code.gs).

function doGet(e) {
  var origin = e ? e.parameter.origin : "";
  return handleResponse({ status: "success", data: getDatabaseData() }, origin);
}

function doPost(e) {
  var origin = e ? e.parameter.origin : "";
  try {
    var payload = JSON.parse(e.postData.contents);
    var action = payload.action;
    var result;

    if (action === "sync_tasks") {
      result = syncTasksToSheet(payload.tasks);
    } else if (action === "sync_comments") {
      result = syncCommentsToSheet(payload.comments);
    } else {
      throw new Error("Unsupported action");
    }

    return handleResponse({ status: "success", result: result }, origin);
  } catch (err) {
    return handleResponse({ status: "error", message: err.toString() }, origin);
  }
}

function handleResponse(data, origin) {
  var output = ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
  
  // Custom CORS response configuration
  return output;
}

function getDatabaseData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // Tasks Sheet Config
  var taskSheet = ss.getSheetByName("Tasks") || ss.insertSheet("Tasks");
  var taskData = [];
  if (taskSheet.getLastRow() > 1) {
    var rows = taskSheet.getRange(2, 1, taskSheet.getLastRow() - 1, 9).getValues();
    taskData = rows.map(function(r) {
      return {
        id: r[0],
        title: r[1],
        description: r[2],
        owner: r[3],
        status: r[4],
        dueDate: r[5],
        priority: r[6],
        createdAt: r[7],
        updatedAt: r[8]
      };
    });
  } else {
    // Set headers if brand new
    taskSheet.getRange(1, 1, 1, 9).setValues([["id", "title", "description", "owner", "status", "dueDate", "priority", "createdAt", "updatedAt"]]);
  }

  // Comments Sheet Config
  var commentSheet = ss.getSheetByName("Comments") || ss.insertSheet("Comments");
  var commentData = [];
  if (commentSheet.getLastRow() > 1) {
    var rows = commentSheet.getRange(2, 1, commentSheet.getLastRow() - 1, 5).getValues();
    commentData = rows.map(function(r) {
      return {
        id: r[0],
        taskId: r[1],
        author: r[2],
        commentText: r[3],
        timestamp: r[4]
      };
    });
  } else {
    commentSheet.getRange(1, 1, 1, 5).setValues([["id", "taskId", "author", "commentText", "timestamp"]]);
  }

  return { tasks: taskData, comments: commentData };
}

function syncTasksToSheet(tasks) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Tasks");
  sheet.clearContents();
  sheet.getRange(1, 1, 1, 9).setValues([["id", "title", "description", "owner", "status", "dueDate", "priority", "createdAt", "updatedAt"]]);
  
  if (tasks && tasks.length > 0) {
    var values = tasks.map(function(t) {
      return [
        t.id,
        t.title,
        t.description,
        t.owner,
        t.status,
        t.dueDate,
        t.priority,
        t.createdAt,
        t.updatedAt
      ];
    });
    sheet.getRange(2, 1, values.length, 9).setValues(values);
  }
  return "Tasks Synchronized successfully. Size: " + (tasks ? tasks.length : 0);
}

function syncCommentsToSheet(comments) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Comments");
  sheet.clearContents();
  sheet.getRange(1, 1, 1, 5).setValues([["id", "taskId", "author", "commentText", "timestamp"]]);

  if (comments && comments.length > 0) {
    var values = comments.map(function(c) {
      return [
        c.id,
        c.taskId,
        c.author,
        c.commentText,
        c.timestamp
      ];
    });
    sheet.getRange(2, 1, values.length, 5).setValues(values);
  }
  return "Comments Synchronized successfully. Size: " + (comments ? comments.length : 0);
}
